<!-- Load Netlify Identity Widget -->
<script src="https://identity.netlify.com/v1/netlify-identity-widget.js"></script>

<!-- Login + Redirect Logic -->
<script>
  (function () {
    function init() {
      if (!window.netlifyIdentity) return;

      // If already logged in, redirect right away
      const user = window.netlifyIdentity.currentUser();
      if (user) {
        window.location.assign('/');
        return;
      }

      // Button opens login modal
      const btn = document.getElementById('open-login');
      if (btn) btn.addEventListener('click', () => window.netlifyIdentity.open('login'));

      // Open login automatically when page loads
      window.netlifyIdentity.on('init', () => window.netlifyIdentity.open('login'));

      // Redirect after successful login
      window.netlifyIdentity.on('login', () => {
        window.location.assign('/');
      });

      // Optional: handle logout
      window.netlifyIdentity.on('logout', () => {
        console.log('User logged out');
      });

      // Optional: handle errors
      window.netlifyIdentity.on('error', (e) => {
        console.error('Identity error:', e);
        alert('Unable to sign in right now. Please try again.');
      });

      window.netlifyIdentity.init();
    }

    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
  })();
</script>
