# Sign in

Welcome back to **PlanIt** ✨

Click the button below to sign in.  
Once signed in, you’ll be redirected to the homepage.

<button id="open-login" style="
  padding:.75rem 1rem;
  border-radius:12px;
  border:0;
  font-weight:600;
  cursor:pointer;
  background:#4f46e5;
  color:#fff;
  margin-top: 1rem;
">
  Open Sign-in
</button>

<p style="margin-top:.75rem; color:#666">
  Don’t have an account yet? <a href="/signup/">Create one</a>.
</p>

<script src="https://identity.netlify.com/v1/netlify-identity-widget.js"></script>

<script>
  (function () {
    function init() {
      if (!window.netlifyIdentity) return;

      // If already logged in, redirect right away
      var user = window.netlifyIdentity.currentUser();
      if (user) {
        window.location.assign('/');
        return;
      }

      // Button opens login modal
      var btn = document.getElementById('open-login');
      if (btn) btn.addEventListener('click', function () {
        window.netlifyIdentity.open('login');
      });

      // Open login automatically when page loads
      window.netlifyIdentity.on('init', function () {
        window.netlifyIdentity.open('login');
      });

      // Redirect after successful login
      window.netlifyIdentity.on('login', function () {
        window.location.assign('/');
      });

      // Optional: handle logout
      window.netlifyIdentity.on('logout', function () {
        console.log('User logged out');
      });

      // Optional: handle errors
      window.netlifyIdentity.on('error', function (e) {
        console.error('Identity error:', e);
        alert('Unable to sign in right now. Please try again.');
      });

      window.netlifyIdentity.init();
    }

    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
  })();
</script>
